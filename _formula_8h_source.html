145 FormulaTree* buildTree (const std::string& expression)
146 {
147 //Create new node of tree
148 auto formula = Parser::removeSpace (expression);
149 FormulaTree* node = new FormulaTree (formula);
150
151 // If it has parentheses then we eliminate them and go to the next node
152 if (formula[0] == '(' && formula[int(formula.size() - 1)] == ')') {
153 if (Parser::isNextParenthesesValid (formula)) {
154 node->parentheses = true;
155 node->leftTree = buildTree (deleteParentheses (formula));
156 return node;
157 }
158 }
159
160 // Check if the input is a function
161 node->function = Parser::parseNextFunction (formula);
162
163 if (node->function != Parser::Function::None) {
164 const auto functionSize = Parser::toString (node->function).size () + 1;
165 if (node->function == Parser::Function::sqrt) {
166 node->leftTree = buildTree (deleteParentheses (formula.substr (functionSize)));
167 }
168 else {
169 node->leftTree = buildTree (formula.substr (functionSize));
170 }
171 return node;
172 }
173
174 // Check what is the lowest priority operation in the current string. Addition
175 int addSubIndex = -1;
176 int addSubPriority = 1e9;
177 // Multiplication
178 int mulIndex = -1;
179 int mulPriority = 1e9;
180 // Division
181 int divIndex = -1;
182 int divPriority = 1e9;
183 // Power
184 int powIndex = -1;
185 int powPriority = 1e9;
186 // Group
187 int minimumPriority = 1e9;
188 int currentPriority = 0;
189
190 int index = 0;
191 for (const auto character : formula) {
192 // Group
193 if (character == '(') currentPriority++;
194 if (character == ')') currentPriority--;
195 // Addition
196 if ((character == '+' || character == '-') && currentPriority < addSubPriority) {
197 addSubPriority = currentPriority;
198 addSubIndex = index;
199 }
200 // Multiplication
201 if ((character == '*' || character == sgui::Parser::dot) && currentPriority < mulPriority) {
202 mulPriority = currentPriority;
203 mulIndex = index;
204 }
205 // Division
206 if ((character == '/') && currentPriority < divPriority) {
207 divPriority = currentPriority;
208 divIndex = index;
209 }
210 // Power
211 if ((character == '^') && currentPriority < powPriority) {
212 powPriority = currentPriority;
213 powIndex = index;
214 }
215 index++;
216 }
217
218 // Check if there is at least an operator
219 if (addSubIndex == -1 && mulIndex == -1 && powIndex == -1 && divIndex == -1) {
220 return node;
221 }
222
223 // Find which operator has the lowest priority and it's location
224 minimumPriority = std::min (std::min (addSubPriority, mulPriority), std::min(powPriority, divPriority));
225
226 int operatorIndex = 0;
227 if (minimumPriority == addSubPriority) {
228 operatorIndex = addSubIndex;
229 }
230 else if (minimumPriority == mulPriority) {
231 operatorIndex = mulIndex;
232 }
233 else if (minimumPriority == divPriority) {
234 operatorIndex = divIndex;
235 }
236 else {
237 operatorIndex = powIndex;
238 }
239
240 // Break the current formula in two parts -> the left part of the operation and the right part
241 node->operation = formula [operatorIndex];
242 node->leftTree = buildTree (formula.substr (0, operatorIndex));
243 node->rightTree = buildTree (formula.substr (operatorIndex + 1));
244 return node;
245 }

249 void buildFormulaCoordinates (FormulaTree* node, float heightPos, float& widthPos)
250 {
251 // Quit if there is no formula
252 if (node == nullptr) return;
253
254 // Check if node is a leaf and mark the coordinates
255 if (isOnlyTerm (node)) {
256 node->position.xFormula = widthPos;
257 node->position.yFormula = heightPos;
258 node->height.up = heightPos;
259 node->height.down = heightPos;
260 node->size.width = getWidthOfString (node->formula);
261 node->size.height = 0;
262 widthPos += getWidthOfString (node->formula);
263 return;
264 }
265
266 if (node->operation == '/') {
267 // Get height and width coordinates
268 const float savedWidthPos = widthPos;
269 buildFormulaCoordinates (node->leftTree, heightPos - 0.7f * characterSize, widthPos);
270 widthPos = savedWidthPos;
271 buildFormulaCoordinates (node->rightTree, heightPos + 0.7f * characterSize, widthPos);
272 widthPos = savedWidthPos;
273
274 // Rebuild the position with correct settings
275 auto maxWidth = std::max (node->leftTree->size.width, node->rightTree->size.width);
276 widthPos += (maxWidth - node->leftTree->size.width) * 0.5f;
277 buildFormulaCoordinates (node->leftTree, (heightPos - 0.7f * characterSize) * 2 - node->leftTree->height.down, widthPos);
278
279 maxWidth = std::max (node->leftTree->size.width, node->rightTree->size.width);
280 widthPos = savedWidthPos + (maxWidth - node->rightTree->size.width) * 0.5f;
281 buildFormulaCoordinates (node->rightTree, (heightPos + 0.7f * characterSize) * 2 - node->rightTree->height.up + characterSize * 0.28f, widthPos);
282
283 node->height.up = node->leftTree->height.up;
284 node->height.down = node->rightTree->height.down;
285 node->size.width = maxWidth;
286 node->size.height = std::abs (node->height.down - node->height.up);
287
288 node->position.xOperator = savedWidthPos;
289 node->position.yOperator = heightPos + 0.7f * characterSize;
290
291 //Reset position after fraction
292 widthPos = savedWidthPos + std::max (node->leftTree->size.width, node->rightTree->size.width);
293 return;
294 }
295 else if (node->operation == '^') {
296 // Get base coordinates
297 buildFormulaCoordinates(node->leftTree, heightPos, widthPos);
298 const float savedWidthPos = widthPos;
299
300 // Calculate the proportion of the exponent
301 buildFormulaCoordinates(node->rightTree, heightPos - characterSize*0.7, widthPos);
302 widthPos = savedWidthPos;
303
304 // Recalculate the position depending on the size of the exponent
305 buildFormulaCoordinates(node->rightTree, (heightPos - characterSize*0.7) * 2 - node->rightTree->height.down - node->leftTree->size.height, widthPos);
306
307 // Save the information of the current formula
308 node->height.up = node->rightTree->height.up;
309 node->height.down = node->leftTree->height.down;
310 node->size.width = node->leftTree->size.width + node->rightTree->size.width;
311 node->size.height = std::max(0.f, node->height.down - node->height.up);
312 return;
313 }
314 else if (node->parentheses) {
315 node->position.xOperator = widthPos;
316 // Calculate the proportion of the formula within the parantheses
317 buildFormulaCoordinates (node->leftTree, heightPos, widthPos);
318
319 widthPos = node->position.xOperator + getWidthOfChar ('(', node->leftTree->size.height)
320 + getHeightOfChar ('(', node->leftTree->size.height) * 0.15f;
321
322 // Recalculate the positions
323 buildFormulaCoordinates (node->leftTree, heightPos, widthPos);
324
325 // Adjust height of parantheses depending on the proportions of the formula
326 node->position.yOperator = (node->leftTree->height.down + node->leftTree->height.up
327 - std::abs(node->leftTree->height.down - node->leftTree->height.up)) * 0.5f
328 - (node->leftTree->size.height > 0 ? (getHeightOfChar ('(', node->leftTree->size.height) * 0.25f) : 0)
329 + (node->leftTree->operation == '^' ? characterSize * 0.7f : 0);
330 widthPos = widthPos + getHeightOfChar (')', node->leftTree->size.height) * 0.15f + getWidthOfChar(')', node->leftTree->size.height);
331
332 // Save the information of the current formula
333 node->size.width = widthPos - node->position.xOperator;
334 node->height.up = node->leftTree->height.up;
335 node->height.down = node->leftTree->height.down;
336 node->size.height = std::abs (node->height.up - node->height.down);
337 return;
338 }
339 else if (node->function != Parser::Function::None) {
340 node->position.xOperator = widthPos;
341 node->position.yOperator = heightPos;
342 widthPos += getWidthOfString (Parser::toString (node->function));
343
344 if (node->function == Parser::Function::sqrt) {
345 // widthPos = node->position.xOperator + getWidthOfString ("LL");
346 buildFormulaCoordinates (node->leftTree, heightPos + characterSize * 0.2f, widthPos);
347 }
348 else {
349 buildFormulaCoordinates (node->leftTree, heightPos, widthPos);
350 }
351
352 node->size.width = widthPos - node->position.xOperator;
353 node->size.height = node->leftTree->size.height;
354 node->height.up = node->leftTree->height.up - characterSize * 0.20f;
355 node->height.down = node->leftTree->height.down;
356 return;
357 }
358 else if (!node->parentheses) {
359 // Build the first term
360 buildFormulaCoordinates(node->leftTree, heightPos, widthPos);
361
362 // Build the operator
363 widthPos += spacing;
364 node->position.xOperator = widthPos;
365 node->position.yOperator = heightPos;
366 widthPos += (spacing + getWidthOfChar(node->operation));
367
368 //Build the second term
369 buildFormulaCoordinates(node->rightTree, heightPos, widthPos);
370
371 //Save the information of the current formula
372 node->height.down = std::max (node->leftTree->height.down, node->rightTree->height.down);
373 node->height.up = std::min (node->leftTree->height.up, node->rightTree->height.up);
374 node->size.width =
375 node->leftTree->size.width + node->rightTree->size.width + 2 * spacing + getWidthOfChar (node->operation);
376 node->size.height = std::abs (node->height.down - node->height.up);
377 return;
378 }

382 void drawFormula (FormulaTree* node, sf::RenderWindow& window)
383 {
384 // quit if node is empty
385 if (node == nullptr) return;
386
387 // set global text properties
388 auto text = sf::Text (mathFont);
389 text.setCharacterSize (characterSize);
390 text.setFillColor (color);
391
392 // Base node
393 if (isOnlyTerm (node)) {
394 text.setPosition ({node->position.xFormula, node->position.yFormula});
395 text.setString (node->formula);
396 window.draw (text);
397 return;
398 }
399 // Parentheses
400 else if (node->parentheses) {
401 // left parenthesis
402 text.setCharacterSize (characterSize + node->leftTree->size.height * 1.2f);
403 text.setPosition ({node->position.xOperator, node->position.yOperator});
404 text.setString ("(");
405 window.draw (text);
406
407 // formula
408 drawFormula (node->leftTree, window);
409
410 // right parenthesis
411 text.setString (")");
412 const auto x = node->position.xOperator
413 + node->leftTree->size.width
414 + 0.15f * getHeightOfChar (')', node->leftTree->size.height)
415 + 0.609f * getWidthOfChar (')', node->leftTree->size.height)
416 + int(characterSize * 3 / getWidthOfChar (')', node->leftTree->size.height));
417 text.setPosition ({x, node->position.yOperator});
418 window.draw (text);
419 return;
420 }
421 // Functions
422 else if (node->function != Parser::Function::None) {
423 if (node->function == Parser::Function::sqrt) {
424 text.setPosition ({node->position.xOperator, node->position.yOperator});
425 text.setCharacterSize (characterSize + node->leftTree->size.height * 1.2f);
426 text.setString (Parser::sqrt);
427 window.draw (text);
428 }
429 else {
430 text.setPosition ({node->position.xOperator, node->position.yOperator});
431 text.setString (Parser::toString (node->function));
432 window.draw (text);
433 }
434 drawFormula (node->leftTree, window);
435 return;
436 }
437 // Operations
438 else if (node->operation == '^') {
439 drawFormula (node->leftTree, window);
440 drawFormula (node->rightTree, window);
441 }
442 else if (node->operation == '/') {
443 drawFormula (node->leftTree, window);
444 sf::RectangleShape line ({node->size.width, characterSize * 0.1f});
445 line.setPosition ({node->position.xOperator, node->position.yOperator});
446 line.setFillColor (color);
447 window.draw (line);
448 drawFormula (node->rightTree, window);
449 }
450 else if (node->operation != '^' && node->operation != '/') {
451 text.setPosition ({node->position.xOperator, node->position.yOperator});
452 text.setString (node->operation);
453 drawFormula (node->leftTree, window);
454 window.draw (text);
455 drawFormula (node->rightTree, window);
456 }
457 }
